# epp-registrobr
Modulo para integração epp registrobr.

## Sample
```
var Epp = require('registrobr-epp');
var fs = require('fs');
var connection;

connection = new Epp({
  login: "<login>",
  password: "<password>"
});

connection.hello().then((data) => {
  console.log(data);
}).catch((err) => {
  console.log(err);
});
```

## Configs
```
{
  "ssl": <ssl>,
  "key": "<key>",
  "cert": "<cert>",
  "ca": ["<ca>"],
  "port": <port>,

  "password": "<password>"
```
